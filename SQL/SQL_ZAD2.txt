SELECT I.SUMMARY
	,I.DESCRIPTION
	,L.NAME
FROM ISSUESS I
	,ISSUESLISTES L
WHERE I.ISSUELIST_ID = L.ID;

SELECT I.SUMMARY
	,I.DESCRIPTION
	,U.FIRSTNAME
	,U.LASTNAME
FROM ISSUESS I
	,USERS U
WHERE I.USER_ID_ASSIGNEDTO= U.ID;

SELECT U.FIRSTNAME
	,U.LASTNAME
	,COUNT(*) AS COUNT
FROM ISSUESS I
	,USERS U
WHERE I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY U.FIRSTNAME
	,U.LASTNAME
ORDER BY U.FIRSTNAME
	,U.LASTNAME
