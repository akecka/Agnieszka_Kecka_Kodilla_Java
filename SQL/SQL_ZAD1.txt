CREATE TABLE ISSUESLISTES
(
	ID SERIAL PRIMARY KEY,
    NAME VARCHAR(150)
);

CREATE TABLE ISSUESS
(
	ID SERIAL PRIMARY KEY,
    ISSUELIST_ID BIGINT UNSIGNED,
    SUMMARY VARCHAR(100),
    DESCRIPTION VARCHAR(100),
    USER_ID_ASSIGNEDTO BIGINT UNSIGNED,
    FOREIGN KEY(USER_ID_ASSIGNEDTO) REFERENCES USERS(ID),
    FOREIGN KEY(ISSUELIST_ID) REFERENCES ISSUESLISTES(ID)
);
